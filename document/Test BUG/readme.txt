select AUFNR, MATERIAL as Item, RESERVATION_NUMBER as ReserveNo, RESERVATION_ITEM as ReserveLine, SUM(REQ_QUAN) as qty into #temp1 from sconit5_si.dbo.SI_SAP_ProdOrderBomDet where  AUFNR in ('000007231712', '000007231585')
group by AUFNR, MATERIAL, RESERVATION_NUMBER, RESERVATION_ITEM;

drop table #temp2;
select b.AUFNR, b.Item, b.ReserveNo, b.ReserveLine, SUM(OrderQty) as Qty into #temp2 from ORD_OrderBomDet as b inner join ORD_OrderMstr_4 as m on m.OrderNo = b.OrderNo 
where m.ExtOrderNo = '000007231585' group by b.AUFNR, b.Item, b.ReserveNo, b.ReserveLine

drop table #temp3;
select AUFNR, Item, ReserveNo, ReserveLine, SUM(OrderQty) as Qty into #temp3 from VIEW_OrderDet as d
inner join VIEW_OrderMstr as m on d.OrderNo = m.OrderNo where m.ExtOrderNo = '000007231585' and Flow like 'R%'
and d.AUFNR is not null
group by  AUFNR, Item, ReserveNo, ReserveLine

drop table #temp4;
select a.AUFNR, a.Item, a.ReserveNo, a.ReserveLine, sum(a.Qty) as Qty into #temp4
from
(select * from #temp2
union all
select * from #temp3)  as a
--where AUFNR = '000007231585' and Item = '5801271261' and ReserveNo = '0000536325' and ReserveLine = '0949'
group by a.AUFNR, a.Item, a.ReserveNo, a.ReserveLine

select * from #temp1 as a left join #temp4 as b
on a.AUFNR = b.AUFNR 
and a.Item = b.Item and a.ReserveNo = b.ReserveNo and a.ReserveLine = b.ReserveLine
where a.Qty <> b.qty or b.Qty is null
